318  public Point ScreenToPoint(double x, double y)
319  {
320  double pointX;
321  double pointY;
322  double halfCell = this.cellSize / 2;
323  double centerX = x + this.cam.X;
324  double isoY = y + this.cam.Y;
325  double isoX = centerX + halfCell - (this.windowWidth / 2);
326  pointX = (Math.Floor(isoX / halfCell) + Math.Floor(isoY / (halfCell / 2))) / 2;
327  pointY = (Math.Floor(isoY / (halfCell / 2)) - Math.Floor(isoX / halfCell)) / 2;
328  return new Point(pointX, pointY);
329  }
