336  public WareType SetType(int x, int y)
337  {
338  Dictionary<WareType, int> counter = new Dictionary<WareType, int>();
339  counter.Add(WareType.Goods, 0);
340  counter.Add(WareType.Mail, 0);
341  counter.Add(WareType.Nothing, 0);
342  counter.Add(WareType.Ore, 0);
343 
344  for (int i = -2; i < 3; i++)
345  {
346  for (int j = -2; j < 3; j++)
347  {
348  if (!(i == 0 && j == 0))
349  {
350  MapItem item = this.GetBuildingByCoord(x + i, y + j);
351  if (item != null)
352  {
353  if (item is House)
354  {
355  counter[WareType.Mail]++;
356  }
357  else if (item is Factory)
358  {
359  counter[WareType.Goods]++;
360  }
361  else if (item is Refinery)
362  {
363  counter[WareType.Ore]++;
364  }
365  }
366  else
367  {
368  counter[WareType.Nothing]++;
369  }
370  }
371  }
372  }
373 
374  if (!(counter[WareType.Mail] > 0 || counter[WareType.Goods] > 0 || counter[WareType.Ore] > 0))
375  {
376  return WareType.Nothing;
377  }
378  else if (counter[WareType.Goods] >= counter[WareType.Mail] && counter[WareType.Goods] >= counter[WareType.Ore])
379  {
380  return WareType.Goods;
381  }
382  else if (counter[WareType.Ore] >= counter[WareType.Mail] && counter[WareType.Ore] >= counter[WareType.Goods])
383  {
384  return WareType.Ore;
385  }
386  else
387  {
388  return WareType.Mail;
